df = df_agg

df_results = pd.DataFrame({'metrics':['Accuracy', 'Recall', 'KFCV-Mean Accuracy', 
                                      'KFCV-Variance','TN', 'TP', 'FN', 'FP']})

df_fi = pd.DataFrame()

numbers_list = list(range(1, 11))
#list_width = list(np.arange(1, 11, 0.5)/)

for i in numbers_list:
    # Down-sample majority class
    df_majority = df[df.Target == 0]
    df_minority = df[df.Target == 1]
    df_majority_downsampled = resample(df_majority,
                                       replace=False,
                                       n_samples=df_minority.shape[0],
                                       random_state=i)
    df_downsampled = pd.concat([df_majority_downsampled, df_minority])

    features = [
        'Volume_7days', 'Volume_15days', 'Tempo_Medio_Chat',
        'Tempo_Medio_Email', 'AWT_Chat', '%NFCR', '%Insatisfao(CSAT)',
        'CSAT_Rated', 'Plan_value', 'Count_checkin', 'Cancel'
    ]
    
    # Traning and Testing sets
    y = df_downsampled.Target
    X = df_downsampled.filter(features)
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.25, random_state=42)
    
    
    
    # Random Forest model
    # Fitting Random Forest Classification to the Training set
    classifier_rf = RandomForestClassifier(n_estimators=100,
                                           max_depth=6,
                                           criterion='entropy',
                                           random_state=i)
    classifier_rf.fit(X_train, y_train)

    # Predicting the Test set results
    y_pred_rf = classifier_rf.predict(X_test)

    # K-foldcross validation
    from sklearn.model_selection import cross_val_score
    accuracies_rf = cross_val_score(estimator=classifier_rf,
                                    X=X_train,
                                    y=y_train,
                                    cv=10,
                                    n_jobs=-1)
    
    # Evaluations
    tn_rf, fp_rf, fn_rf, tp_rf = confusion_matrix(y_test, y_pred_rf).ravel()
    
    model = 'RF_' + str(i) 
    
    df_results[model] = [accuracy_score(y_test, y_pred_rf), recall_score(y_test, y_pred_rf), 
                         accuracies_rf.mean(), accuracies_rf.std(),
                         tn_rf, tp_rf, fn_rf, fp_rf]
    
    #print(model)
    #print(f'Accuracy = {accuracy_score(y_test, y_pred_rf):.2f} \nRecall = {recall_score(y_test, y_pred_rf):.2f}', '\n')
    
    
    # Features Importances
    fi = pd.DataFrame({model +'_feature': list(X_train.columns),
                   model + '_importance': classifier_rf.feature_importances_}).\
                    sort_values(model +'_feature', ascending = False)
    df_fi = pd.concat([df_fi, fi], axis=1)
    
    
    plt.barh(fi[model +'_feature'],fi[ model + '_importance'], label=model, align='center')
    plt.xlabel("Random Forest Feature Importance")
    plt.title('Model Features Stability')
    plt.legend()
